<script>
function flipCard(deck) {
if (deck.dataset.animating === 'true') return;
const cards = deck.querySelectorAll('.content-card');
if (cards.length < 2) return;
deck.dataset.animating = 'true';
const topCard = cards[0];
topCard.classList.add('flipping');
setTimeout(() => {
topCard.classList.remove('flipping');
deck.appendChild(topCard);
updateCardZIndex(deck);
deck.dataset.animating = 'false';
}, 400);
}
function updateCardZIndex(deck) {
const cards = deck.querySelectorAll('.content-card');
cards.forEach((card, index) => {
card.style.zIndex = cards.length - index;
if (index === 0) {
card.style.transform = 'translate(0, 0)';
card.style.opacity = '1';
card.classList.add('active-card');
} else {
card.style.transform = `translate(${index * 6}px, ${index * 6}px)`;
card.style.opacity = '1';
card.classList.remove('active-card');
}
});
}
document.querySelectorAll('.cards-deck').forEach(updateCardZIndex);
</script>
