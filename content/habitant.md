// ➡️ 往後翻頁 (點擊右側 70% 區域)
function nextCard(deck) {
if (deck.dataset.animating === 'true') return;
const cards = deck.querySelectorAll('.content-card');
if (cards.length < 2) return;
deck.dataset.animating = 'true';

const topCard = cards[0];
topCard.classList.add('flipping-forward');

setTimeout(() => {
topCard.classList.remove('flipping-forward');
deck.appendChild(topCard); // 把第一張放到最後面
updateCardZIndex(deck);
deck.dataset.animating = 'false';
}, 400);
}

// ⬅️ 往前翻頁 (點擊左側 30% 區域)
function prevCard(deck) {
if (deck.dataset.animating === 'true') return;
const cards = deck.querySelectorAll('.content-card');
if (cards.length < 2) return;
deck.dataset.animating = 'true';

const lastCard = cards[cards.length - 1];
lastCard.classList.add('flipping-backward');
deck.insertBefore(lastCard, cards[0]); // 把最後一張抽到最前面
updateCardZIndex(deck);

setTimeout(() => {
lastCard.classList.remove('flipping-backward');
deck.dataset.animating = 'false';
}, 400);
}

// 🧮 重新動態計算景深與 zIndex
function updateCardZIndex(deck) {
const cards = deck.querySelectorAll('.content-card');
cards.forEach((card, index) => {
card.style.zIndex = cards.length - index;
if (index === 0) {
card.style.transform = 'translate(0, 0) scale(1)';
card.style.opacity = '1';
card.style.pointerEvents = 'auto';
card.classList.add('active-card');
} else {
card.style.transform = `translate(${index * 6}px, ${index * 6}px) scale(1)`;
card.style.opacity = index < 3 ? '1' : '0'; // 只顯示前三張堆疊感，其餘隱藏防混亂
card.style.pointerEvents = 'none';
card.classList.remove('active-card');
}
});
}

// 初始化全站牌堆
document.querySelectorAll('.cards-deck').forEach(updateCardZIndex);

// Intersection Observer 滾動側邊導覽監控
const observerOptions = { root: null, rootMargin: '0px', threshold: 0.6 };
const observer = new IntersectionObserver((entries) => {
entries.forEach(entry => {
if (entry.isIntersecting) {
const id = entry.target.getAttribute('id');
document.querySelectorAll('.habitant-quick-nav a').forEach(navLink => navLink.classList.remove('active'));
const activeLink = document.querySelector(`.habitant-quick-nav a[href="#${id}"]`);
if (activeLink) { activeLink.classList.add('active'); }
}
});
}, observerOptions);

document.querySelectorAll('.deck-container').forEach(section => observer.observe(section));
</script>
